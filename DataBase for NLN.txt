CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    address TEXT,

    create_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    update_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE doctors (
    doctor_id INT NOT NULL PRIMARY KEY,
    specialization VARCHAR(255) NOT NULL,
    clinic_room VARCHAR(50),
    working_time VARCHAR(255),
    experience_years INT,
    introduction TEXT,
    FOREIGN KEY (doctor_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE appointments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_id INT NOT NULL,        -- ID người bệnh (tham chiếu tới users.id)
    doctor_id INT NOT NULL,         -- ID bác sĩ (tham chiếu tới doctors.id)
    appointment_date DATE NOT NULL, -- Ngày khám
    session ENUM('Sáng', 'Chiều') NOT NULL, -- Buổi khám
    phone VARCHAR(20),              -- Số điện thoại liên hệ (có thể lấy từ users hoặc ghi lại khi đặt)
    symptoms TEXT NOT NULL,         -- Triệu chứng mô tả
    follow_up BOOLEAN DEFAULT FALSE, -- Có tái khám hay không
    status ENUM('pending', 'confirmed', 'completed', 'cancelled') DEFAULT 'pending', -- Trạng thái lịch hẹn
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (patient_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (doctor_id) REFERENCES doctors(doctor_id) ON DELETE CASCADE
);


CREATE TABLE medicines (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,             -- Tên thuốc
    description TEXT,                       -- Mô tả thêm (công dụng, thành phần,...)
    dosage_form VARCHAR(100),               -- Dạng thuốc (viên nén, dung dịch, xịt mũi,...)
    strength VARCHAR(100),                  -- Hàm lượng (vd: 500mg, 5mg/ml,...)
    unit VARCHAR(50),                       -- Đơn vị (hộp, lọ, viên,...)
    price DECIMAL(10,2) NOT NULL DEFAULT 0, -- Giá bán/đơn vị
    available BOOLEAN DEFAULT TRUE          -- Còn hàng hay không
);

CREATE TABLE medical_history (
    id INT AUTO_INCREMENT PRIMARY KEY,
    appointment_id INT NOT NULL,     -- liên kết tới lịch hẹn
    patient_id INT NOT NULL,         -- bệnh nhân
    doctor_id INT NOT NULL,          -- bác sĩ
    diagnosis TEXT NOT NULL,         -- chẩn đoán của bác sĩ
    treatment_plan TEXT,             -- kế hoạch điều trị
    notes TEXT,                     -- ghi chú bổ sung
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (appointment_id) REFERENCES appointments(id) ON DELETE CASCADE,
    FOREIGN KEY (patient_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (doctor_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE medical_history_medicines (
    id INT AUTO_INCREMENT PRIMARY KEY,
    medical_history_id INT NOT NULL, -- liên kết tới lịch sử bệnh án
    medicine_id INT NOT NULL,        -- thuốc kê
    quantity INT NOT NULL,            -- số lượng
    dosage VARCHAR(255),              -- liều dùng (vd: "uống 2 viên/ngày")
    notes TEXT,                      -- ghi chú thêm về thuốc

    FOREIGN KEY (medical_history_id) REFERENCES medical_history(id) ON DELETE CASCADE,
    FOREIGN KEY (medicine_id) REFERENCES medicines(id) ON DELETE RESTRICT
);

CREATE TABLE services (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,       -- tên dịch vụ
    description TEXT,                 -- mô tả chi tiết dịch vụ
    price DECIMAL(10, 2) NOT NULL,   -- giá dịch vụ
    is_active BOOLEAN DEFAULT TRUE,  -- trạng thái còn hoạt động hay không
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
