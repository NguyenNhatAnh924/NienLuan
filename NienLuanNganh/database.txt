-- Tạo CSDL nếu chưa tồn tại
CREATE DATABASE IF NOT EXISTS clinic_management;

-- Sử dụng CSDL vừa tạo
USE clinic_management;


-- Xóa bảng nếu đã tồn tại để tránh lỗi khi chạy lại script
DROP TABLE IF EXISTS medical_history, appointment, receptionist, patient, admin, doctor, users;

-- Bảng users (thêm cột age)
CREATE TABLE users (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    address VARCHAR(255),
    role ENUM('doctor', 'admin', 'patient', 'receptionist') DEFAULT 'patient',
    age INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
ALTER TABLE users MODIFY email VARCHAR(100) NULL;


-- Bảng doctor
CREATE TABLE doctor (
    user_id INT UNSIGNED PRIMARY KEY,
    specialization VARCHAR(255),
    license_number VARCHAR(255),
    experience FLOAT,
    sex VARCHAR(10),
    level VARCHAR(50),
    degree VARCHAR(255),
    department VARCHAR(255),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Thêm lương và trạng thái cho doctor
ALTER TABLE doctor
ADD COLUMN salary FLOAT DEFAULT 0,
ADD COLUMN employment_status ENUM('working', 'resigned', 'retired') DEFAULT 'working';


-- Bảng admin
CREATE TABLE admin (
    user_id INT UNSIGNED PRIMARY KEY,
    role ENUM('super_admin', 'admin') DEFAULT 'admin',
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
ALTER TABLE admin
ADD COLUMN salary FLOAT DEFAULT 0,
ADD COLUMN employment_status VARCHAR(50) DEFAULT '',
ADD COLUMN position VARCHAR(100) DEFAULT '',
ADD COLUMN experience FLOAT DEFAULT 0,
ADD COLUMN sex VARCHAR(10) DEFAULT '',
ADD COLUMN level VARCHAR(50) DEFAULT '',
ADD COLUMN degree VARCHAR(100) DEFAULT '';

-- Bảng patient
CREATE TABLE patient (
    user_id INT UNSIGNED PRIMARY KEY,
    medical_history TEXT,
    date_of_birth DATE,
    blood_type VARCHAR(10),
    insurance_number VARCHAR(255),
    allergies TEXT,
    emergency_contact_name VARCHAR(100),
    emergency_contact_phone VARCHAR(20),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Bảng receptionist
CREATE TABLE receptionist (
    user_id INT UNSIGNED PRIMARY KEY,
    department VARCHAR(255),
    shift_time VARCHAR(255),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Thêm lương và trạng thái cho receptionist
ALTER TABLE receptionist
ADD COLUMN salary FLOAT DEFAULT 0,
ADD COLUMN employment_status ENUM('Đang làm việc', 'Đã nghỉ việc', 'Nghĩ hưu') DEFAULT 'Đang làm việc';
ALTER TABLE receptionist
ADD COLUMN degree VARCHAR(255) DEFAULT '',
ADD COLUMN level VARCHAR(255) DEFAULT '';
ALTER TABLE receptionist
ADD COLUMN sex VARCHAR(10) DEFAULT '';

-- Bảng appointment
CREATE TABLE appointment (
    appointment_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    patient_id INT UNSIGNED NOT NULL,
    doctor_id INT UNSIGNED NOT NULL,
    appointment_date DATETIME NOT NULL,
    status ENUM('pending', 'completed', 'cancelled') NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (patient_id) REFERENCES patient(user_id) ON DELETE CASCADE,
    FOREIGN KEY (doctor_id) REFERENCES doctor(user_id) ON DELETE CASCADE
);
ALTER TABLE appointment ADD COLUMN description TEXT AFTER status;


-- Bảng medical_history
CREATE TABLE medical_history (
    history_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    patient_id INT UNSIGNED NOT NULL,
    doctor_id INT UNSIGNED NOT NULL,
    appointment_id INT UNSIGNED, -- Cho phép NULL vì ON DELETE SET NULL
    diagnosis TEXT,
    treatment_plan TEXT,
    follow_up DATETIME,
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (patient_id) REFERENCES patient(user_id) ON DELETE CASCADE,
    FOREIGN KEY (doctor_id) REFERENCES doctor(user_id) ON DELETE CASCADE,
    FOREIGN KEY (appointment_id) REFERENCES appointment(appointment_id) ON DELETE SET NULL
);

-- Xóa bảng nếu đã tồn tại để tránh lỗi khi chạy lại script
DROP TABLE IF EXISTS testresult;

-- Tạo bảng testresult liên kết với medical_history
CREATE TABLE testresult (
    testResultID INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    history_id INT UNSIGNED NOT NULL, -- Liên kết với bảng medical_history
    testResultName VARCHAR(255) NOT NULL, -- Tên xét nghiệm
    testDate DATETIME NOT NULL, -- Ngày làm xét nghiệm
    resultDate DATETIME NOT NULL, -- Ngày trả kết quả
    conclusion TEXT, -- Kết luận
    notes TEXT, -- Ghi chú
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (history_id) REFERENCES medical_history(history_id) ON DELETE CASCADE
);

-- Bảng invoices (Hóa đơn)
CREATE TABLE invoices (
    invoices_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    paymentStatus BOOLEAN NOT NULL DEFAULT 0
);

-- Bảng services (Dịch vụ bệnh viện)
CREATE TABLE services (
    serviceId INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    serviceName VARCHAR(255) NOT NULL,
    servicePrice FLOAT NOT NULL,
    serviceDescription TEXT
);

-- Bảng invoices_details (Chi tiết hóa đơn dịch vụ)
CREATE TABLE invoices_details (
    invoicesDetailsId INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    invoices_id INT UNSIGNED NOT NULL,
    serviceId INT UNSIGNED NOT NULL,
    FOREIGN KEY (invoices_id) REFERENCES invoices(invoices_id) ON DELETE CASCADE,
    FOREIGN KEY (serviceId) REFERENCES services(serviceId) ON DELETE CASCADE
);

-- Bảng Medication (Danh sách thuốc)
CREATE TABLE Medication (
    medication_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    medication_name VARCHAR(255) NOT NULL,
    manufacturer VARCHAR(255),
    price FLOAT NOT NULL,
    description TEXT
);

-- Bảng prescription (Đơn thuốc)
CREATE TABLE prescription (
    prescription_id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    prescription_name VARCHAR(255) NOT NULL,
    dosage_instructions TEXT
);

-- Bảng details_prescription (Chi tiết đơn thuốc)
CREATE TABLE details_prescription (
    detailsPrescriptionID INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    prescription_id INT UNSIGNED NOT NULL,
    medication_id INT UNSIGNED NOT NULL,
    prescribedDose INT NOT NULL,
    FOREIGN KEY (prescription_id) REFERENCES prescription(prescription_id) ON DELETE CASCADE,
    FOREIGN KEY (medication_id) REFERENCES Medication(medication_id) ON DELETE CASCADE
);
